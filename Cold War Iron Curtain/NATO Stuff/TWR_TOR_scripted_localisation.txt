defined_text = {
	name = GetTORMilitaryGFX
	text = {
		trigger = {
			if = {
				limit = { 
					check_variable = { TOR.TOR_military_cooperation < 20 }
				}
				set_temp_variable = { TOR_military_1_frame = 6 }
			}
			else = { 
				set_temp_variable = { TOR_military_1_frame = 1 }
			}

			if = {
				limit = { 
					check_variable = { TOR.TOR_military_cooperation < 40 }
				}
				set_temp_variable = { TOR_military_2_frame = 7 }
			}
			else = { 
				set_temp_variable = { TOR_military_2_frame = 2 }
			}

			if = {
				limit = { 
					check_variable = { TOR.TOR_military_cooperation < 50 }
				}
				set_temp_variable = { TOR_military_3_frame = 8 }
			}
			else = { 
				set_temp_variable = { TOR_military_3_frame = 3 }
			}

			if = {
				limit = { 
					check_variable = { TOR.TOR_military_cooperation < 75 }
				}
				set_temp_variable = { TOR_military_4_frame = 9 }
			}
			else = { 
				set_temp_variable = { TOR_military_4_frame = 4 }
			}

			if = {
				limit = { 
					check_variable = { TOR.TOR_military_cooperation < 90 }
				}
				set_temp_variable = { TOR_military_5_frame = 10 }
			}
			else = { 
				set_temp_variable = { TOR_military_5_frame = 5 }
			}
		}
		localization_key = "TOR_military_operability_bar_bg"
	}
}
