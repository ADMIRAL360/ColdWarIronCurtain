country_event = {
	id = twrta.1011
	title = twrta.1011.t
	desc = twrta.1011.d
	picture = GFX_report_event_summit
	hidden = yes 

	is_triggered_only = yes
	fire_only_once = no
	
	immediate = {
		log = "calculations for TOR cycling exercises"

		TOR = {

			set_variable = { TOR_exercise_engagement = TOR_total_exercise_involvement }
			divide_variable = { TOR_exercise_engagement = TOR.TOR_member_countries^num }

			if = {
				limit = {
					check_variable = { TOR_exercise_engagement > 1 }
				}

				if = {
					limit = {
						check_variable = { TOR_exercise_engagement > 2 }
					}
					add_to_variable = { TOR.TOR_military_cooperation = 1 }
				}
				else = {
					add_to_variable = { TOR.TOR_military_cooperation = 0.5 }
				}

				for_each_scope_loop = {
					array = TOR.TOR_member_countries
	
					country_event = {
						id = twrta.1012 # exercise concluded successfully
						random_days = 5 
					}
				}
			}
			else = {
				for_each_scope_loop = {
					array = TOR.TOR_member_countries
	
					country_event = {
						id = twrta.1013 # exercise concluded, but not enough participation
						random_days = 5 
					}
				}

				add_to_variable = { TOR.TOR_military_cooperation = -2 }
			}
		}
	}
}
