on_actions = {

	on_startup = {
		effect = {
			TOR = {
				add_to_array = { TOR_member_countries = USA }
				add_to_array = { TOR_member_countries = ENG }
				add_to_array = { TOR_member_countries = CAN }
				add_to_array = { TOR_member_countries = AST }
				add_to_array = { TOR_member_countries = NZL }
				add_to_array = { TOR_member_countries = ICE }

				set_variable = { TOR_military_cooperation = 45 }
				set_variable = { TOR_political_integration = 20 }

				country_event = {
					id = twrta.1005
				}
			}
		}
	}
	on_monthly_USA = {
		effect = {
			if = {
				limit = {
					NOT = {
						check_variable = { TOR.TOR_political_integration < 20 }
					}
				}
				add_to_variable = { TOR.TOR_political_integration = 0.25 }
				add_to_variable = { TOR.TOR_military_cooperation = 0.25 }
			}
		}
	}
	on_daily_USA = {
		effect = {
			TOR = {
				TOR_recalculate_member_effects = yes 
			}
		}
	}

	on_join_faction = {
		effect = {
			if = {
				limit = { 
					FROM = {
						TAG = TOR 
					}
				 }

				TOR_apply_military_effects = yes 
				TOR_apply_political_effects = yes 
			}
		}
	}
	on_leave_faction = {
		effect = {
			if = {
				limit = { 
					FROM = {
						TAG = TOR 
					}
				}
				TOR_remove_all_effects = yes 
			}
		}
	}
}
