#Step Towards Left
india_economic_towards_left = {
    custom_effect_tooltip = shift_economic_system_left_TT
    if = {
        limit = {
            check_variable = {
                var_economic_system_food_consumption > 0
            }
        }
        add_to_variable = {
            var_economic_system_food_consumption = -0.0125
        }
    }
    if = {
        limit = {
            check_variable = {
                var_economic_system_electricity_consumption > 0
            }
        }
        add_to_variable = {
            var_economic_system_electricity_consumption = -0.0125
        }
    }
    if = {
        limit = {
            check_variable = {
                research_speed_factor > 0
            }
            add_to_variable = {
                research_speed_factor = -0.005
            }
        }
    }
    add_to_variable = {
        var_economic_system_office_park_income_adjustment = -0.0125
    }
    add_to_variable = {
        var_economic_system_agri_production_bonus = 0.0125
    }
    add_to_variable = {
        economic_system_agri_speed = 0.0125
    }
    add_to_variable = {
        economic_system_industrial_speed = 0.0125
    }
    add_to_variable = {
        economic_system_office_park_speed = -0.0125
    }
    add_to_variable = {
        var_consumer_goods_need = -0.0125
    }
}
